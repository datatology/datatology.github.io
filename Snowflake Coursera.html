RETURNS BOOLEAN

LANGUAGE SQL

AS $$

DECLARE

ts1 TIMESTAMP;

ts2 TIMESTAMP;

BEGING

ts1 := (SELECT MAX(order_ts) FROM t1);

DELETE FROM t1 WHERE ts < :ts1;

END;
